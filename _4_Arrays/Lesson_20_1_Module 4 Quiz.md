# A numeric array stores each array element with a numeric index.
- True
--------------------------------------------------------
# Which statement will be displayed in the browser?
<?php 
$array[0] = "the mall";
$array[1] = "David";
$array[2] = "brother";
$array[3] = "the store";
$array[4] = "Rob";
echo "$array[1] went to $array[3]";
?>

- David went to the store
